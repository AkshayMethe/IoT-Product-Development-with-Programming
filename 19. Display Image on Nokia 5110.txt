Nokia5110LCD::Display nokiaLCD(D0, D1, D2, A0);

void lcdFunTime();

void setup()
{

	pinMode(D7, OUTPUT);

	nokiaLCD.begin(); // This will setup our pins, and initialize the LCD
	nokiaLCD.setContrast(60); // Pretty good value, play around with it

	 for (int i=10;i>0;i--){
            digitalWrite(D7, HIGH);
            delay(i*i*3);
            digitalWrite(D7, LOW);
            delay(i*i*2);
        }
	
	nokiaLCD.clearDisplay(WHITE); // Begin by clearing the display

}

void loop()
{
	nokiaLCD.setBitmap(AIM1);
	nokiaLCD.updateDisplay();
	delay(2500);

	nokiaLCD.setBitmap(AIM2);
	nokiaLCD.updateDisplay();
	delay(100);

	nokiaLCD.setBitmap(AIM3);
	nokiaLCD.updateDisplay();
	delay(100);

	nokiaLCD.setBitmap(AIM4);
	nokiaLCD.updateDisplay();
	delay(100);
	
	nokiaLCD.setBitmap(AIM5);
	nokiaLCD.updateDisplay();
	delay(100);

	nokiaLCD.setBitmap(AIM6);
	nokiaLCD.updateDisplay();
	delay(100);
	
	//After frame 6 the logo moves off screen, so we get 1 frame with a white clear display.
	nokiaLCD.clearDisplay(WHITE);
	nokiaLCD.updateDisplay();
	delay(500);
	
	nokiaLCD.setBitmap(AIM7);
	nokiaLCD.updateDisplay();
	delay(100);

	nokiaLCD.setBitmap(AIM8);
	nokiaLCD.updateDisplay();
	delay(100);

	nokiaLCD.setBitmap(AIM9);

	nokiaLCD.setBitmap(AIM10);

	delay(100);
	
	nokiaLCD.setBitmap(AIM11);
	nokiaLCD.updateDisplay();
	delay(100);

}
